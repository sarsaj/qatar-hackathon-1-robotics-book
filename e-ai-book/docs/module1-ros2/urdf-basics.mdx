## Code Examples

```xml
<!-- Example URDF snippet for a simple robot arm link -->
<link name="base_link">
  <visual>
    <geometry>
      <box size="0.6 0.4 0.1" />
    </geometry>
    <material name="blue">
      <color rgba="0 0 0.8 1" />
    </material>
  </visual>
  <collision>
    <geometry>
      <box size="0.6 0.4 0.1" />
    </geometry>
  </collision>
  <inertial>
    <origin xyz="0 0 0" rpy="0 0 0" />
    <mass value="1.0" />
    <inertia ixx="0.001" ixy="0" ixz="0" iyy="0.001" iyz="0" izz="0.001" />
  </inertial>
</link>

<joint name="base_to_arm_joint" type="revolute">
  <parent link="base_link" />
  <child link="arm_link" />
  <origin xyz="0 0 0.1" rpy="0 0 0" />
  <axis xyz="0 0 1" />
  <limit lower="-1.57" upper="1.57" velocity="0.5" effort="10" />
</joint>
```
